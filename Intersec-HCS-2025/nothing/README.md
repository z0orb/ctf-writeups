def decrypt():
    # Create reverse lookup dictionary for more efficient decoding
    reverse_sbox = {v: k for k, v in SBox.items()}
    
    # Split the flag by tab character (each encoded character ends with \t)
    encoded_chars = flag.split('\t')[:-1]  # Remove last empty element from split
    
    # Decode each character
    decoded_chars = []
    for encoded_char in encoded_chars:
        # Add back the tab to match SBox format
        encoded_with_tab = encoded_char + '\t'
        
        # Look up the original number
        if encoded_with_tab in reverse_sbox:
            original_num = reverse_sbox[encoded_with_tab]
            # Convert number to ASCII character
            decoded_chars.append(chr(original_num))
    
    # Join all decoded characters to form the flag
    decoded_flag = ''.join(decoded_chars)
    print(decoded_flag)
